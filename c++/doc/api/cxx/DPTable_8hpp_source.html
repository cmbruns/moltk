00107     void compute_recurrence(const DPCell& up_left,
00108                             const DPCell& up,
00109                             const DPCell& left,
00110                             const DPPosition<SCORE_TYPE, GAP_NSEGS>& pos1,
00111                             const DPPosition<SCORE_TYPE, GAP_NSEGS>& pos2) // score for insertion gap in sequence 2
00112     {
00113         // This recurrence comes from Gusfield chapter 11.
00114         // best score with ungapped alignment of i with j
00115         g.score = up_left.v.score + pos1.score(pos2);
00116         // best score with insertion gap in sequence 1

00118         // best score with insertion gap in sequence 2
